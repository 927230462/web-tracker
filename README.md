# web-tracker


# 安装
npm install web-tracker --save-dev


# 使用
```javascript
import jsTracker from 'web-tracker' //引入
jsTracker.init({  //初始化
      host: 'cn-shanghai.log.aliyuncs.com', //必填
      project: 'web-tracker-test', //必填
      logstore: 'web-tracker-test-store', //必填
      pcode: "TEST", //必填
      version: "1.0.0", //必填
})
```
